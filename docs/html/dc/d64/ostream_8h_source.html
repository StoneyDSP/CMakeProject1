94// Write the content of buf to os.
95// It is a separate function rather than a part of vprint to simplify testing.
96template <typename Char>
97void write_buffer(std::basic_ostream<Char>& os, buffer<Char>& buf) {
98 if (const_check(FMT_MSC_VERSION)) {
99 auto filebuf = dynamic_cast<std::basic_filebuf<Char>*>(os.rdbuf());
100 if (filebuf && write(*filebuf, {buf.data(), buf.size()})) return;
101 }
102 const Char* buf_data = buf.data();
103 using unsigned_streamsize = std::make_unsigned<std::streamsize>::type;
104 unsigned_streamsize size = buf.size();
105 unsigned_streamsize max_size = to_unsigned(max_value<std::streamsize>());
106 do {
107 unsigned_streamsize n = size <= max_size ? size : max_size;
108 os.write(buf_data, static_cast<std::streamsize>(n));
109 buf_data += n;
110 size -= n;
111 } while (size != 0);
112}

114template <typename Char, typename T>
115void format_value(buffer<Char>& buf, const T& value,
116 locale_ref loc = locale_ref()) {
117 auto&& format_buf = formatbuf<std::basic_streambuf<Char>>(buf);
118 auto&& output = std::basic_ostream<Char>(&format_buf);
119#if !defined(FMT_STATIC_THOUSANDS_SEPARATOR)
120 if (loc) output.imbue(loc.get<std::locale>());
121#endif
122 output << value;
123 output.exceptions(std::ios_base::failbit | std::ios_base::badbit);
124}

130// Formats an object of type T that has an overloaded ostream operator<<.
131template <typename Char>
132struct basic_ostream_formatter : formatter<basic_string_view<Char>, Char> {
133 template <typename T, typename OutputIt>
134 auto format(const T& value, basic_format_context<OutputIt, Char>& ctx) const
135 -> OutputIt {
136 auto buffer = basic_memory_buffer<Char>();
137 format_value(buffer, value, ctx.locale());
138 return formatter<basic_string_view<Char>, Char>::format(
139 {buffer.data(), buffer.size()}, ctx);
140 }
141};
