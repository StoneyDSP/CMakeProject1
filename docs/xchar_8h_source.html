202inline void vprint(std::FILE* f, wstring_view fmt, wformat_args args) {
203 wmemory_buffer buffer;
204 detail::vformat_to(buffer, fmt, args);
205 buffer.push_back(L'\0');
206 if (std::fputws(buffer.data(), f) == -1)
207 FMT_THROW(system_error(errno, FMT_STRING("cannot write to file")));
208}
